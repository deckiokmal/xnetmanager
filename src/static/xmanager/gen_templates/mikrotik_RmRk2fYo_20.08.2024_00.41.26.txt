/ip route
add dst-address=0.0.0.0/0 gateway=ISP1-IP routing-mark=to_ISP1
add dst-address=0.0.0.0/0 gateway=ISP2-IP routing-mark=to_ISP2 

/ip firewall mangle
add action=mark-routing chain=prerouting dst-address-type=!local in-interface=Local new-routing-mark=to_ISP1 passthrough=yes
add action=mark-routing chain=prerouting dst-address-type=!local in-interface=Local new-routing-mark=to_ISP2 passthrough=yes

/ip firewall nat
add chain=srcnat out-interface=ISP1-Interface action=masquerade
add chain=srcnat out-interface=ISP2-Interface action=masquerade
